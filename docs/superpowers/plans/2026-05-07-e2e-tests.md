**Goal:** Add a Playwright E2E test suite that exercises the full multiplayer game flow with 3+ players per test across 10 scenarios.

**Architecture:** Multiple `browserContext` instances (one per player) within a single Playwright process. Playwright auto-starts `pnpm dev` via `webServer` config with `ROUND_DELAY_MS=1000` to keep round transitions fast. Guest display names (no auth required).

**Tech Stack:** `@playwright/test`, Chromium, TanStack Start dev server on port 3000, Postgres + Redis via Docker.

```ts
import { expect, type Page } from "@playwright/test";

export async function createGame(
  page: Page,
  options: {
    displayName?: string;
    totalRounds?: number;
    houseRules?: {
      randoCardrissian?: boolean;
      happyEnding?: boolean;
      packingHeat?: boolean;
    };
  } = {}
): Promise<{ roomCode: string; playerId: string }> {
  const {
    displayName = `guest_host_${Date.now()}`,
    totalRounds = 2,
    houseRules = {},
  } = options;

  await page.goto("/games/create");
  // First text input is "Your display name" (no htmlFor on label)
  await page.locator("input[type='text'],input:not([type])").first().fill(displayName);
  // First number input is "Rounds"
  await page.locator("input[type='number']").first().fill(String(totalRounds));

  if (houseRules.randoCardrissian) {
    await page.getByLabel("Rando Cardrissian").check();
  }
  if (houseRules.happyEnding) {
    await page.getByLabel("Happy Ending").check();
  }
  if (houseRules.packingHeat) {
    await page.getByLabel("Packing Heat").check();
  }

  await page.getByRole("button", { name: "Create Game" }).click();
  await page.waitForURL(/\/games\/[A-Z0-9]+\/lobby/, { timeout: 15_000 });

  const url = new URL(page.url());
  const roomCode = url.pathname.split("/")[2];
  const playerId = url.searchParams.get("playerId") ?? "";
  return { roomCode, playerId };
}

export async function joinGame(
  page: Page,
  roomCode: string,
  displayName: string,
  options: { spectator?: boolean } = {}
): Promise<{ playerId: string }> {
  await page.goto("/games/join");
  await page.locator("input[type='text'],input:not([type])").first().fill(displayName);
  await page
    .locator("input[type='text'],input:not([type])")
    .nth(1)
    .fill(roomCode);

  if (options.spectator) {
    await page.getByRole("button", { name: "Spectator" }).click();
  }

  await page.getByRole("button", { name: "Join Game" }).click();
  await page.waitForURL(/\/games\/[A-Z0-9]+\/(lobby|session)/, {
    timeout: 15_000,
  });

  const url = new URL(page.url());
  const playerId = url.searchParams.get("playerId") ?? "";
  return { playerId };
}

export async function startGame(hostPage: Page): Promise<void> {
  await hostPage.getByRole("button", { name: /Start Game/ }).click();
}

export async function waitForSession(page: Page): Promise<void> {
  await page.waitForURL(/\/games\/[A-Z0-9]+\/session/, { timeout: 20_000 });
}

export async function findCzar(pages: Page[]): Promise<{
  czarPage: Page;
  nonCzarPages: Page[];
}> {
  const flags = await Promise.all(
    pages.map((p) =>
      p.getByText(/Czar — wait for submissions/).isVisible()
    )
  );
  const czarIdx = flags.indexOf(true);
  if (czarIdx === -1) throw new Error("No czar found among pages");
  return {
    czarPage: pages[czarIdx],
    nonCzarPages: pages.filter((_, i) => i !== czarIdx),
  };
}

// Returns null for czarPage when Rando Cardrissian is czar (no human controls them)
export async function findCzarOrNull(pages: Page[]): Promise<{
  czarPage: Page | null;
  nonCzarPages: Page[];
}> {
  const flags = await Promise.all(
    pages.map((p) =>
      p.getByText(/Czar — wait for submissions/).isVisible()
    )
  );
  const czarIdx = flags.indexOf(true);
  if (czarIdx === -1) {
    // Rando is czar — all human pages are non-czars
    return { czarPage: null, nonCzarPages: pages };
  }
  return {
    czarPage: pages[czarIdx],
    nonCzarPages: pages.filter((_, i) => i !== czarIdx),
  };
}

export async function playCard(page: Page): Promise<void> {
  await page
    .locator('[data-testid="hand-card"]:not([data-played="true"])')
    .first()
    .click();
}

export async function pickWinner(czarPage: Page): Promise<void> {
  await czarPage.locator('[data-testid="submission"]').first().click();
}

export async function waitForRound(page: Page, num: number): Promise<void> {
  await expect(page.locator('[data-testid="round-display"]')).toContainText(
    `Round ${num}/`,
    { timeout: 15_000 }
  );
}

export async function getHandSize(page: Page): Promise<number> {
  return page.locator('[data-testid="hand-card"]').count();
}

export async function getScores(
  page: Page
): Promise<Record<string, number>> {
  const entries = await page
    .locator('[data-testid="score-panel-entry"]')
    .all();
  const result: Record<string, number> = {};
  for (const entry of entries) {
    const name = (await entry.getAttribute("data-player")) ?? "";
    const score = Number((await entry.getAttribute("data-score")) ?? "0");
    if (name) result[name] = score;
  }
  return result;
}

export async function playRound(
  pages: Page[]
): Promise<void> {
  const { czarPage, nonCzarPages } = await findCzar(pages);
  await Promise.all(nonCzarPages.map((p) => playCard(p)));
  await expect(
    czarPage.locator('[data-testid="submission"]')
  ).toHaveCount(nonCzarPages.length, { timeout: 15_000 });
  await pickWinner(czarPage);
}
```

```ts
  browser,

  const ts = Date.now();
  const [ctx1, ctx2, ctx3] = await Promise.all([

    browser.newContext(),

      totalRounds: 2,
    });

    await Promise.all([
      joinGame(p2, roomCode, `guest_p2_${ts}`),

    // All players see all 3 in their own lobby

      3,
      { timeout: 10_000 }
    );

    // Host starts — all navigate to session
    await startGame(p1);
    await Promise.all([

      waitForSession(p2),
```

```ts
    for (const p of nonCzarPages) {
```

```ts
  const [p1, p2, p3] = await Promise.all([
    ctx1.newPage(),
    ctx2.newPage(),
    ctx3.newPage(),
  ]);

  try {
    const { roomCode } = await createGame(p1, {
      displayName: `guest_host_${ts}`,
      totalRounds: 2,
    });
    await Promise.all([
      joinGame(p2, roomCode, `guest_p2_${ts}`),
      joinGame(p3, roomCode, `guest_p3_${ts}`),
    ]);

    await expect(p1.locator('[data-testid="player-list-item"]')).toHaveCount(3, { timeout: 10_000 });

    // Register WS listeners on ALL pages BEFORE navigation to session.
    // Playwright's page.on('websocket') only captures connections opened after the listener
    // is registered, so this must happen before startGame triggers the session navigation.
    const allPlayedCounts = new Map<Page, number>([[p1, 0], [p2, 0], [p3, 0]]);
    for (const page of [p1, p2, p3]) {
      page.on("websocket", (ws) => {
        ws.on("framereceived", ({ payload }) => {
          try {
            const msg = JSON.parse(String(payload)) as { event: string };
            if (msg.event === "all:played") {
              allPlayedCounts.set(page, (allPlayedCounts.get(page) ?? 0) + 1);
            }
          } catch {}
        });
      });
    }

    await startGame(p1);
    await Promise.all([waitForSession(p1), waitForSession(p2), waitForSession(p3)]);

    await waitForRound(p1, 1);
    const pages = [p1, p2, p3];
    const { czarPage, nonCzarPages } = await findCzar(pages);

    // Both non-czars submit simultaneously
    await Promise.all(nonCzarPages.map((p) => playCard(p)));

    // Czar sees exactly 2 submissions — no duplicates, no missing
    await expect(czarPage.locator('[data-testid="submission"]')).toHaveCount(2, { timeout: 15_000 });

    // all:played fired exactly once on each connected client (server publishes once, all peers receive once)
    expect(allPlayedCounts.get(p1)).toBe(1);
    expect(allPlayedCounts.get(p2)).toBe(1);
    expect(allPlayedCounts.get(p3)).toBe(1);

    // Game proceeds normally
    await pickWinner(czarPage);
    await expect(
      czarPage.locator('[data-testid="score-panel-entry"][data-score="1"]')
    ).toHaveCount(1, { timeout: 10_000 });
  } finally {
    await Promise.all([ctx1.close(), ctx2.close(), ctx3.close()]);
  }
});
```
